151     /**
152      * When enabled, program standard and error output will be redirected to the
153      * Maven logger as <i>Info</i> and <i>Error</i> level logs, respectively. If not enabled the
154      * traditional behavior of program output being directed to standard System.out
155      * and System.err is used.<br>
156      * <br>
157      * NOTE: When enabled, to log the program standard out as Maven <i>Debug</i> level instead of
158      * <i>Info</i> level use {@code exec.quietLogs=true}. <br>
159      * <br>
160      * This option can be extremely helpful when combined with multithreaded builds
161      * for two reasons:<br>
162      * <ul>
163      * <li>Program output is suffixed with the owning thread name, making it easier
164      * to trace execution of a specific projects build thread.</li>
165      * <li>Program output will not get jumbled with other maven log messages.</li>
166      * </ul>
167      *
168      * For Example, if using {@code exec:exec} to run a script to echo a count from
169      * 1 to 100 as:
170      *
171      * <pre>
172      * for i in {1..100}
173      * do
174      *   echo "${project.artifactId} - $i"
175      * done
176      * </pre>
177      *
178      * When this script is run multi-threaded on two modules, {@code module1} and
179      * {@code module2}, you might get output such as:
180      *
181      * <pre>
182      * [BuilderThread 1] [INFO] --- exec-maven-plugin:1.6.0:exec (test) @ module1 ---
183      * [BuilderThread 2] [INFO] --- exec-maven-plugin:1.6.0:exec (test) @ module2 ---
184      * ...
185      * module2 - 98
186      * modu
187      * module1 - 97
188      * module1 -
189      * le2 - 9899
190      * ...
191      * </pre>
192      *
193      * With this flag enabled, the output will instead come something similar to:
194      *
195      * <pre>
196      * ...
197      * [Exec Stream Pumper] [INFO] [BuilderThread 2] module2 - 98
198      * [Exec Stream Pumper] [INFO] [BuilderThread 1] module1 - 97
199      * [Exec Stream Pumper] [INFO] [BuilderThread 1] module1 - 98
200      * [Exec Stream Pumper] [INFO] [BuilderThread 2] module2 - 99
201      * ...
202      * </pre>
203      *
204      * NOTE 1: To show the thread in the Maven log, configure the Maven
205      * installations <i>conf/logging/simplelogger.properties</i> option:
206      * {@code org.slf4j.simpleLogger.showThreadName=true}<br>
207      *
208      * NOTE 2: This option is ignored when {@code exec.outputFile} is specified.
209      *
210      * @since 3.0.0
211      * @see java.lang.System#err
212      * @see java.lang.System#in
213      */

324     /**
325      * priority in the execute method will be to use System properties arguments over the pom specification.
326      *
327      * @throws MojoExecutionException if a failure happens
328      */
329     public void execute()
330         throws MojoExecutionException
331     {
332         if ( executable == null )
333         {
334             if (executableDependency == null)
335             {
336                 throw new MojoExecutionException( "The parameter 'executable' is missing or invalid" );
337             }
338 
339             executable = findExecutableArtifact().getFile().getAbsolutePath();
340             getLog().debug( "using executable dependency " + executable);
341         }
342 
343         if ( isSkip() )
344         {
345             getLog().info( "skipping execute as per configuration" );
346             return;
347         }
348 
349         if ( basedir == null )
350         {
351             throw new IllegalStateException( "basedir is null. Should not be possible." );
352         }
353 
354         try
355         {
356 
357             handleWorkingDirectory();
358 
359             String argsProp = getSystemProperty( "exec.args" );
360 
361             List<String> commandArguments = new ArrayList<String>();
362 
363             if ( hasCommandlineArgs() )
364             {
365                 handleCommandLineArgs( commandArguments );
366             }
367             else if ( !StringUtils.isEmpty( argsProp ) )
368             {
369                 handleSystemPropertyArguments( argsProp, commandArguments );
370             }
371             else
372             {
373                 if ( arguments != null )
374                 {
375                     handleArguments( commandArguments );
376                 }
377             }
378 
379             Map<String, String> enviro = handleSystemEnvVariables();
380 
381             CommandLine commandLine = getExecutablePath( enviro, workingDirectory );
382 
383             String[] args = commandArguments.toArray( new String[commandArguments.size()] );
384 
385             commandLine.addArguments( args, false );
386 
387             Executor exec = new DefaultExecutor();
388             if ( this.timeout > 0 )
389             {
390                 exec.setWatchdog( new ExecuteWatchdog( this.timeout ) );
391             }
392             exec.setWorkingDirectory( workingDirectory );
393             fillSuccessCodes( exec );
394 
395             getLog().debug( "Executing command line: " + commandLine );
396 
397             try
398             {
399                 int resultCode;
400                 if ( outputFile != null )
401                 {
402                     if ( !outputFile.getParentFile().exists() && !outputFile.getParentFile().mkdirs() )
403                     {
404                         getLog().warn( "Could not create non existing parent directories for log file: " + outputFile );
405                     }
406 
407                     FileOutputStream outputStream = null;
408                     try
409                     {
410                         outputStream = new FileOutputStream( outputFile );
411 
412                         resultCode = executeCommandLine( exec, commandLine, enviro, outputStream );
413                     }
414                     finally
415                     {
416                         IOUtil.close( outputStream );
417                     }
418                 }
419                 else if (useMavenLogger)
420                 {
421                     getLog().debug("Will redirect program output to Maven logger");
422                     final String parentThreadName = Thread.currentThread().getName();
423                     final String logSuffix = "[" + parentThreadName + "] ";
424                     Invokable<String> mavenOutRedirect = new Invokable<String>()
425                     {
426 
427                         @Override
428                         public void accept(String logMessage)
429                         {
430                             if (quietLogs)
431                             {
432                                 getLog().debug(logSuffix + logMessage);
433                             }
434                             else
435                             {
436                                 getLog().info(logSuffix + logMessage);
437                             }
438                         }
439                     };
440                     Invokable<String> mavenErrRedirect = new Invokable<String>()
441                     {
442 
443                         @Override
444                         public void accept(String logMessage)
445                         {
446                             getLog().error(logSuffix + logMessage);
447                         }
448                     };
449 
450                     try (OutputStream out = new LineRedirectOutputStream(mavenOutRedirect);
451                             OutputStream err = new LineRedirectOutputStream(mavenErrRedirect)) {
452                         resultCode = executeCommandLine(exec, commandLine, enviro, out, err);
453                     }
454                 }
455                 else
456                 {
457                     resultCode = executeCommandLine( exec, commandLine, enviro, System.out, System.err );
458                 }
459 
460                 if ( isResultCodeAFailure( resultCode ) )
461                 {
462                     String message = "Result of " + commandLine.toString() + " execution is: '" + resultCode + "'.";
463                     getLog().error( message );
464                     throw new MojoExecutionException( message );
465                 }
466             }
467             catch ( ExecuteException e )
468             {
469                 if ( exec.getWatchdog() != null && exec.getWatchdog().killedProcess() )
470                 {
471                     final String message = "Timeout. Process runs longer than " + this.timeout + " ms.";
472                     getLog().error( message );
473                     throw new MojoExecutionException( message, e );
474                 }
475                 else
476                 {
477                     getLog().error( "Command execution failed.", e );
478                     throw new MojoExecutionException( "Command execution failed.", e );
479                 }
480             }
481             catch ( IOException e )
482             {
483                 getLog().error( "Command execution failed.", e );
484                 throw new MojoExecutionException( "Command execution failed.", e );
485             }
486 
487             registerSourceRoots();
488         }
489         catch ( IOException e )
490         {
491             throw new MojoExecutionException( "I/O Error", e );
492         }
493     }

495     private Map<String, String> handleSystemEnvVariables()
496         throws MojoExecutionException
497     {
498 
499         Map<String, String> enviro = new HashMap<String, String>();
500         try
501         {
502             Properties systemEnvVars = CommandLineUtils.getSystemEnvVars();
503             for ( Map.Entry<?, ?> entry : systemEnvVars.entrySet() )
504             {
505                 enviro.put( (String) entry.getKey(), (String) entry.getValue() );
506             }
507         }
508         catch ( IOException x )
509         {
510             getLog().error( "Could not assign default system enviroment variables.", x );
511         }
512 
513         if ( environmentVariables != null )
514         {
515             enviro.putAll( environmentVariables );
516         }
517 
518         if ( this.environmentScript != null )
519         {
520             getLog().info( "Pick up external environment script: " + this.environmentScript );
521             Map<String, String> envVarsFromScript = this.createEnvs( this.environmentScript );
522             if ( envVarsFromScript != null )
523             {
524                 enviro.putAll( envVarsFromScript );
525             }
526         }
527 
528         if ( this.getLog().isDebugEnabled() )
529         {
530             Set<String> keys = new TreeSet<String>();
531             keys.addAll( enviro.keySet() );
532             for ( String key : keys )
533             {
534                 this.getLog().debug( "env: " + key + "=" + enviro.get( key ) );
535             }
536         }
537 
538         return enviro;
539     }

583     private void handleCommandLineArgs( List<String> commandArguments )
584         throws MojoExecutionException, IOException
585     {
586         String[] args = parseCommandlineArgs();
587         for ( int i = 0; i < args.length; i++ )
588         {
589             if ( isLongClassPathArgument( args[i] ) )
590             {
591                 // it is assumed that starting from -cp or -classpath the arguments
592                 // are: -classpath/-cp %classpath mainClass
593                 // the arguments are replaced with: -jar $TMP/maven-exec.jar
594                 // NOTE: the jar will contain the classpath and the main class
595                 commandArguments.add( "-jar" );
596                 File tmpFile = createJar( computePath( null ), args[i + 2] );
597                 commandArguments.add( tmpFile.getAbsolutePath() );
598                 i += 2;
599             }
600             else if ( args[i].contains( CLASSPATH_TOKEN ) )
601             {
602                 commandArguments.add( args[i].replace( CLASSPATH_TOKEN, computeClasspathString( null ) ) );
603             }
604             else
605             {
606                 commandArguments.add( args[i] );
607             }
608         }
609     }

611     private void handleArguments( List<String> commandArguments )
612         throws MojoExecutionException, IOException
613     {
614         String specialArg = null;
615 
616         for ( int i = 0; i < arguments.size(); i++ )
617         {
618             Object argument = arguments.get( i );
619 
620             if ( specialArg != null )
621             {
622                 if ( isLongClassPathArgument( specialArg ) && argument instanceof Classpath )
623                 {
624                     // it is assumed that starting from -cp or -classpath the arguments
625                     // are: -classpath/-cp %classpath mainClass
626                     // the arguments are replaced with: -jar $TMP/maven-exec.jar
627                     // NOTE: the jar will contain the classpath and the main class
628                     commandArguments.add( "-jar" );
629 
630                     File tmpFile = createJar( computePath( (Classpath) argument ),
631                                               (String) arguments.get( ++i ) );
632                     commandArguments.add( tmpFile.getAbsolutePath() );
633                 }
634                 else if ( isLongModulePathArgument( specialArg ) && argument instanceof Modulepath )
635                 {
636                     String filePath = new File( buildDirectory, "modulepath" ).getAbsolutePath();
637 
638                     StringBuilder modulePath = new StringBuilder();
639                     modulePath.append( '"' );
640 
641                     for ( Iterator<String> it = computePath( (Modulepath) argument ).iterator(); it.hasNext(); )
642                     {
643                         modulePath.append( it.next().replace( "\\", "\\\\" ) );
644                         if ( it.hasNext() )
645                         {
646                             modulePath.append( File.pathSeparatorChar );
647                         }
648                     }
649 
650                     modulePath.append( '"' );
651 
652                     createArgFile( filePath, Arrays.asList( "-p", modulePath.toString() ) );
653                     commandArguments.add( '@' + filePath );
654                 }
655                 else
656                 {
657                     commandArguments.add( specialArg );
658                 }
659 
660                 specialArg = null;
661 
662                 continue;
663             }
664 
665             if ( argument instanceof Classpath )
666             {
667                 Classpath specifiedClasspath = (Classpath) argument;
668                 commandArguments.add( computeClasspathString( specifiedClasspath ) );
669             }
670             else if ( argument instanceof Modulepath )
671             {
672                 Modulepath specifiedModulepath = (Modulepath) argument;
673                 commandArguments.add( computeClasspathString( specifiedModulepath ) );
674             }
675             else if ( isLongModulePathArgument( specialArg ) || isLongClassPathArgument( specialArg ) )
676             {
677                 specialArg = (String) argument;
678             }
679             else
680             {
681                 commandArguments.add( (String) argument );
682             }
683         }
684     }

686     private void fillSuccessCodes( Executor exec )
687     {
688         if ( successCodes != null && successCodes.length > 0 )
689         {
690             exec.setExitValues( successCodes );
691         }
692     }

694     boolean isResultCodeAFailure( int result )
695     {
696         if ( successCodes == null || successCodes.length == 0 )
697         {
698             return result != 0;
699         }
700         for ( int successCode : successCodes )
701         {
702             if ( successCode == result )
703             {
704                 return false;
705             }
706         }
707         return true;
708     }

710     private boolean isLongClassPathArgument( String arg )
711     {
712         return longClasspath && ( "-classpath".equals( arg ) || "-cp".equals( arg ) );
713     }
714 
715     private boolean isLongModulePathArgument( String arg )
716     {
717         return longModulepath && ( "--module-path".equals( arg ) || "-p".equals( arg ) );

720     /**
721      * Compute the classpath from the specified Classpath. The computed classpath is based on the classpathScope. The
722      * plugin cannot know from maven the phase it is executed in. So we have to depend on the user to tell us he wants
723      * the scope in which the plugin is expected to be executed.
724      *
725      * @param specifiedClasspath Non null when the user restricted the dependencies, <code>null</code> otherwise (the
726      *            default classpath will be used)
727      * @return a platform specific String representation of the classpath
728      */
729     private String computeClasspathString( AbstractPath specifiedClasspath )
730     {
731         List<String> resultList = computePath( specifiedClasspath );
732         StringBuffer theClasspath = new StringBuffer();
733 
734         for ( String str : resultList )
735         {
736             addToClasspath( theClasspath, str );
737         }
738 
739         return theClasspath.toString();
740     }

742     /**
743      * Compute the classpath from the specified Classpath. The computed classpath is based on the classpathScope. The
744      * plugin cannot know from maven the phase it is executed in. So we have to depend on the user to tell us he wants
745      * the scope in which the plugin is expected to be executed.
746      *
747      * @param specifiedClasspath Non null when the user restricted the dependencies, <code>null</code> otherwise (the
748      *            default classpath will be used)
749      * @return a list of class path elements
750      */
751     private List<String> computePath( AbstractPath specifiedClasspath )
752     {
753         List<Artifact> artifacts = new ArrayList<>();
754         List<Path> theClasspathFiles = new ArrayList<>();
755         List<String> resultList = new ArrayList<>();
756 
757         collectProjectArtifactsAndClasspath( artifacts, theClasspathFiles );
758 
759         if ( ( specifiedClasspath != null ) && ( specifiedClasspath.getDependencies() != null ) )
760         {
761             artifacts = filterArtifacts( artifacts, specifiedClasspath.getDependencies() );
762         }
763 
764         for ( Path f : theClasspathFiles )
765         {
766             resultList.add( f.toAbsolutePath().toString() );
767         }
768 
769         for ( Artifact artifact : artifacts )
770         {
771             getLog().debug( "dealing with " + artifact );
772             resultList.add( artifact.getFile().getAbsolutePath() );
773         }
774 
775         return resultList;
776     }

779     {
780         if ( theClasspath.length() > 0 )
781         {
782             theClasspath.append( File.pathSeparator );
783         }
784         theClasspath.append( toAdd );

787     private List<Artifact> filterArtifacts( List<Artifact> artifacts, Collection<String> dependencies )
788     {
789         AndArtifactFilter filter = new AndArtifactFilter();
790 
791         filter.add( new IncludesArtifactFilter( new ArrayList<String>( dependencies ) ) ); // gosh
792 
793         List<Artifact> filteredArtifacts = new ArrayList<Artifact>();
794         for ( Artifact artifact : artifacts )
795         {
796             if ( filter.include( artifact ) )
797             {
798                 getLog().debug( "filtering in " + artifact );
799                 filteredArtifacts.add( artifact );
800             }
801         }
802         return filteredArtifacts;
803     }

807     CommandLine getExecutablePath( Map<String, String> enviro, File dir )
808     {
809         File execFile = new File( executable );
810         String exec = null;
811         if ( execFile.isFile() )
812         {
813             getLog().debug( "Toolchains are ignored, 'executable' parameter is set to " + executable );
814             exec = execFile.getAbsolutePath();
815         }
816 
817         if ( exec == null )
818         {
819             Toolchain tc = getToolchain();
820 
821             // if the file doesn't exist & toolchain is null, the exec is probably in the PATH...
822             // we should probably also test for isFile and canExecute, but the second one is only
823             // available in SDK 6.
824             if ( tc != null )
825             {
826                 getLog().info( "Toolchain in exec-maven-plugin: " + tc );
827                 exec = tc.findTool( executable );
828             }
829             else
830             {
831                 if ( OS.isFamilyWindows() )
832                 {
833                     List<String> paths = this.getExecutablePaths( enviro );
834                     paths.add( 0, dir.getAbsolutePath() );
835 
836                     exec = findExecutable( executable, paths );
837                 }
838             }
839         }
840 
841         if ( exec == null )
842         {
843             exec = executable;
844         }
845 
846         CommandLine toRet;
847         if ( OS.isFamilyWindows() && !hasNativeExtension( exec ) && hasExecutableExtension( exec ) )
848         {
849             // run the windows batch script in isolation and exit at the end
850             final String comSpec = System.getenv( "ComSpec" );
851             toRet = new CommandLine( comSpec == null ? "cmd" : comSpec );
852             toRet.addArgument( "/c" );
853             toRet.addArgument( exec );
854         }
855         else
856         {
857             toRet = new CommandLine( exec );
858         }
859 
860         return toRet;
861     }

863     static String findExecutable( final String executable, final List<String> paths )
864     {
865         File f = null;
866         search: for ( final String path : paths )
867         {
868             f = new File( path, executable );
869             if ( !OS.isFamilyWindows() && f.isFile() )
870                 break;
871             else
872                 for ( final String extension : getExecutableExtensions() )
873                 {
874                     f = new File( path, executable + extension );
875                     if ( f.isFile() )
876                         break search;
877                 }
878         }
879 
880         if ( f == null || !f.exists() )
881             return null;
882 
883         return f.getAbsolutePath();
884     }

886     private static boolean hasNativeExtension( final String exec )
887     {
888         final String lowerCase = exec.toLowerCase();
889         return lowerCase.endsWith( ".exe" ) || lowerCase.endsWith( ".com" );
890     }

892     private static boolean hasExecutableExtension( final String exec )
893     {
894         final String lowerCase = exec.toLowerCase();
895         for ( final String ext : getExecutableExtensions() )
896             if ( lowerCase.endsWith( ext ) )
897                 return true;
898 
899         return false;
900     }

906                         : Arrays.asList( StringUtils.split( pathExt.toLowerCase(), File.pathSeparator ) );

941     protected int executeCommandLine( Executor exec, final CommandLine commandLine, Map<String, String> enviro,
942                                       final PumpStreamHandler psh )
943                                           throws ExecuteException, IOException
944     {
945         exec.setStreamHandler( psh );
946 
947         int result;
948         try
949         {
950             psh.start();
951             if ( async )
952             {
953                 if ( asyncDestroyOnShutdown )
954                 {
955                     exec.setProcessDestroyer( getProcessDestroyer() );
956                 }
957 
958                 exec.execute( commandLine, enviro, new ExecuteResultHandler()
959                 {
960                     public void onProcessFailed( ExecuteException e )
961                     {
962                         getLog().error( "Async process failed for: " + commandLine, e );
963                     }
964 
965                     public void onProcessComplete( int exitValue )
966                     {
967                         getLog().info( "Async process complete, exit value = " + exitValue + " for: " + commandLine );
968                         try
969                         {
970                             psh.stop();
971                         }
972                         catch ( IOException e )
973                         {
974                             getLog().error( "Error stopping async process stream handler for: " + commandLine, e );
975                         }
976                     }
977                 } );
978                 result = 0;
979             }
980             else
981             {
982                 result = exec.execute( commandLine, enviro );
983             }
984         }
985         finally
986         {
987             if ( !async )
988             {
989                 psh.stop();
990             }
991         }
992         return result;
993     }

1077     /**
1078      * Create a jar with just a manifest containing a Main-Class entry for SurefireBooter and a Class-Path entry for all
1079      * classpath elements. Copied from surefire (ForkConfiguration#createJar())
1080      *
1081      * @param classPath List&lt;String> of all classpath elements.
1082      * @return
1083      * @throws IOException
1084      */
1085     private File createJar( List<String> classPath, String mainClass )
1086         throws IOException
1087     {
1088         File file = File.createTempFile( "maven-exec", ".jar" );
1089         file.deleteOnExit();
1090         FileOutputStream fos = new FileOutputStream( file );
1091         JarOutputStream jos = new JarOutputStream( fos );
1092         jos.setLevel( JarOutputStream.STORED );
1093         JarEntry je = new JarEntry( "META-INF/MANIFEST.MF" );
1094         jos.putNextEntry( je );
1095 
1096         Manifest man = new Manifest();
1097 
1098         // we can't use StringUtils.join here since we need to add a '/' to
1099         // the end of directory entries - otherwise the jvm will ignore them.
1100         StringBuilder cp = new StringBuilder();
1101         for ( String el : classPath )
1102         {
1103             // NOTE: if File points to a directory, this entry MUST end in '/'.
1104             cp.append( new URL( new File( el ).toURI().toASCIIString() ).toExternalForm() + " " );
1105         }
1106 
1107         man.getMainAttributes().putValue( "Manifest-Version", "1.0" );
1108         man.getMainAttributes().putValue( "Class-Path", cp.toString().trim() );
1109         man.getMainAttributes().putValue( "Main-Class", mainClass );
1110 
1111         man.write( jos );
1112         jos.close();
1113 
1114         return file;
1115     }
