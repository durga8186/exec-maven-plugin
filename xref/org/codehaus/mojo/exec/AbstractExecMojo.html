22  import java.io.File;
23  import java.util.List;
24  
25  import org.apache.maven.artifact.Artifact;
26  import org.apache.maven.plugin.AbstractMojo;
27  import org.apache.maven.plugin.MojoExecutionException;
28  import org.apache.maven.plugins.annotations.Parameter;
29  import org.apache.maven.project.MavenProject;
30  import org.codehaus.plexus.util.cli.CommandLineUtils;
31  
32  /**
33   * This class is used for unifying functionality between the 2 mojo exec plugins ('java' and 'exec'). It handles parsing
34   * the arguments and adding source/test folders.
35   *
36   * @author Philippe Jacot (PJA)
37   * @author Jerome Lacoste
38   */
39  public abstract class AbstractExecMojo
40      extends AbstractMojo
41  {
42      /**
43       * The enclosing project.
44       */
45      @Parameter( defaultValue = "${project}", readonly = true )
46      protected MavenProject project;
47  
48      /**
49       * This folder is added to the list of those folders containing source to be compiled. Use this if your plugin
50       * generates source code.
51       */
52      @Parameter( property = "sourceRoot" )
53      private File sourceRoot;
54  
55      /**
56       * This folder is added to the list of those folders containing source to be compiled for testing. Use this if your
57       * plugin generates test source code.
58       */
59      @Parameter( property = "testSourceRoot" )
60      private File testSourceRoot;
61  
62      /**
63       * Arguments separated by space for the executed program. For example: "-j 20"
64       */
65      @Parameter( property = "exec.args" )
66      private String commandlineArgs;
67  
68      /**
69       * Defines the scope of the classpath passed to the plugin. Set to compile,test,runtime or system depending on your
70       * needs. Since 1.1.2, the default value is 'runtime' instead of 'compile'.
71       */
72      @Parameter( property = "exec.classpathScope", defaultValue = "runtime" )
73      protected String classpathScope;
74  
75      /**
76       * Skip the execution. Starting with version 1.4.0 the former name <code>skip</code> has been changed into
77       * <code>exec.skip</code>.
78       * 
79       * @since 1.0.1
80       */
81      // TODO: Remove the alias in version 1.5.0 of the plugin.
82      @Parameter( property = "exec.skip", defaultValue = "false", alias = "skip" )
83      private boolean skip;
84  
85      /**
86       * Collects the project artifacts in the specified List and the project specific classpath (build output and build
87       * test output) Files in the specified List, depending on the plugin classpathScope value.
88       *
89       * @param artifacts the list where to collect the scope specific artifacts
90       * @param theClasspathFiles the list where to collect the scope specific output directories
91       */
92      @SuppressWarnings( "unchecked" )
93      protected void collectProjectArtifactsAndClasspath( List<Artifact> artifacts, List<File> theClasspathFiles )
94      {
95  
96          if ( "compile".equals( classpathScope ) )
97          {
98              artifacts.addAll( project.getCompileArtifacts() );
99              theClasspathFiles.add( new File( project.getBuild().getOutputDirectory() ) );
100         }
101         else if ( "test".equals( classpathScope ) )
102         {
103             artifacts.addAll( project.getTestArtifacts() );
104             theClasspathFiles.add( new File( project.getBuild().getTestOutputDirectory() ) );
105             theClasspathFiles.add( new File( project.getBuild().getOutputDirectory() ) );
106         }
107         else if ( "runtime".equals( classpathScope ) )
108         {
109             artifacts.addAll( project.getRuntimeArtifacts() );
110             theClasspathFiles.add( new File( project.getBuild().getOutputDirectory() ) );
111         }
112         else if ( "system".equals( classpathScope ) )
113         {
114             artifacts.addAll( project.getSystemArtifacts() );
115         }
116         else
117         {
118             throw new IllegalStateException( "Invalid classpath scope: " + classpathScope );
119         }
120 
121         getLog().debug( "Collected project artifacts " + artifacts );
122         getLog().debug( "Collected project classpath " + theClasspathFiles );
123     }
124 
125     /**
126      * Parses the argument string given by the user. Strings are recognized as everything between STRING_WRAPPER.
127      * PARAMETER_DELIMITER is ignored inside a string. STRING_WRAPPER and PARAMETER_DELIMITER can be escaped using
128      * ESCAPE_CHAR.
129      *
130      * @return Array of String representing the arguments
131      * @throws MojoExecutionException for wrong formatted arguments
132      */
133     protected String[] parseCommandlineArgs()
134         throws MojoExecutionException
135     {
136         if ( commandlineArgs == null )
137         {
138             return null;
139         }
140         else
141         {
142             try
143             {
144                 return CommandLineUtils.translateCommandline( commandlineArgs );
145             }
146             catch ( Exception e )
147             {
148                 throw new MojoExecutionException( e.getMessage() );
149             }
150         }
151     }
152 
153     /**
154      * @return true of the mojo has command line arguments
155      */
156     protected boolean hasCommandlineArgs()
157     {
158         return ( commandlineArgs != null );
159     }
160 
161     /**
162      * Register compile and compile tests source roots if necessary
163      */
164     protected void registerSourceRoots()
165     {
166         if ( sourceRoot != null )
167         {
168             getLog().info( "Registering compile source root " + sourceRoot );
169             project.addCompileSourceRoot( sourceRoot.toString() );
170         }
171 
172         if ( testSourceRoot != null )
173         {
174             getLog().info( "Registering compile test source root " + testSourceRoot );
175             project.addTestCompileSourceRoot( testSourceRoot.toString() );
176         }
177     }
178 
179     /**
180      * Check if the execution should be skipped
181      *
182      * @return true to skip
183      */
184     protected boolean isSkip()
185     {
186         return skip;
187     }
188 }
