119     /**
120      * Whether to interrupt/join and possibly stop the daemon threads upon quitting. <br/>
121      * If this is <code>false</code>, maven does nothing about the daemon threads. When maven has no more work to do,
122      * the VM will normally terminate any remaining daemon threads.
123      * <p>
124      * In certain cases (in particular if maven is embedded), you might need to keep this enabled to make sure threads
125      * are properly cleaned up to ensure they don't interfere with subsequent activity. In that case, see
126      * {@link #daemonThreadJoinTimeout} and {@link #stopUnresponsiveDaemonThreads} for further tuning.
127      * </p>
128      * 
129      * @since 1.1-beta-1
130      */
131     @Parameter( property = "exec.cleanupDaemonThreads", defaultValue = "true" )
132     private boolean cleanupDaemonThreads;
133 
134     /**
135      * This defines the number of milliseconds to wait for daemon threads to quit following their interruption.<br/>
136      * This is only taken into account if {@link #cleanupDaemonThreads} is <code>true</code>. A value &lt;=0 means to
137      * not timeout (i.e. wait indefinitely for threads to finish). Following a timeout, a warning will be logged.
138      * <p>
139      * Note: properly coded threads <i>should</i> terminate upon interruption but some threads may prove problematic: as
140      * the VM does interrupt daemon threads, some code may not have been written to handle interruption properly. For
141      * example java.util.Timer is known to not handle interruptions in JDK &lt;= 1.6. So it is not possible for us to
142      * infinitely wait by default otherwise maven could hang. A sensible default value has been chosen, but this default
143      * value <i>may change</i> in the future based on user feedback.
144      * </p>
145      * 
146      * @since 1.1-beta-1
147      */
148     @Parameter( property = "exec.daemonThreadJoinTimeout", defaultValue = "15000" )
149     private long daemonThreadJoinTimeout;
150 
151     /**
152      * Wether to call {@link Thread#stop()} following a timing out of waiting for an interrupted thread to finish. This
153      * is only taken into account if {@link #cleanupDaemonThreads} is <code>true</code> and the
154      * {@link #daemonThreadJoinTimeout} threshold has been reached for an uncooperative thread. If this is
155      * <code>false</code>, or if {@link Thread#stop()} fails to get the thread to stop, then a warning is logged and
156      * Maven will continue on while the affected threads (and related objects in memory) linger on. Consider setting
157      * this to <code>true</code> if you are invoking problematic code that you can't fix. An example is
158      * {@link java.util.Timer} which doesn't respond to interruption. To have <code>Timer</code> fixed, vote for
159      * <a href="http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6336543">this bug</a>.
160      * 
161      * @since 1.1-beta-1
162      */
163     @Parameter( property = "exec.stopUnresponsiveDaemonThreads", defaultValue = "false" )
164     private boolean stopUnresponsiveDaemonThreads;

186     /**
187      * Execute goal.
188      * 
189      * @throws MojoExecutionException execution of the main class or one of the threads it generated failed.
190      * @throws MojoFailureException something bad happened...
191      */
192     public void execute()
193         throws MojoExecutionException, MojoFailureException
194     {
195         if ( isSkip() )
196         {
197             getLog().info( "skipping execute as per configuration" );
198             return;
199         }
200         if ( killAfter != -1 )
201         {
202             getLog().warn( "Warning: killAfter is now deprecated. Do you need it ? Please comment on MEXEC-6." );
203         }
204 
205         if ( null == arguments )
206         {
207             arguments = new String[0];
208         }
209 
210         if ( getLog().isDebugEnabled() )
211         {
212             StringBuffer msg = new StringBuffer( "Invoking : " );
213             msg.append( mainClass );
214             msg.append( ".main(" );
215             for ( int i = 0; i < arguments.length; i++ )
216             {
217                 if ( i > 0 )
218                 {
219                     msg.append( ", " );
220                 }
221                 msg.append( arguments[i] );
222             }
223             msg.append( ")" );
224             getLog().debug( msg );
225         }
226 
227         IsolatedThreadGroup threadGroup = new IsolatedThreadGroup( mainClass /* name */ );
228         Thread bootstrapThread = new Thread( threadGroup, new Runnable()
229         {
230             public void run()
231             {
232                 int sepIndex = mainClass.indexOf( '/' );
233 
234                 final String bootClassName;
235                 if ( sepIndex >= 0 )
236                 {
237                     bootClassName = mainClass.substring( sepIndex + 1 );
238                 }
239                 else 
240                 {
241                     bootClassName = mainClass;
242                 }
243                 
244                 try
245                 {
246                     Class<?> bootClass = Thread.currentThread().getContextClassLoader().loadClass( bootClassName );
247                     
248                     MethodHandles.Lookup lookup = MethodHandles.lookup();
249 
250                     MethodHandle mainHandle =
251                         lookup.findStatic( bootClass, "main",
252                                                  MethodType.methodType( void.class, String[].class ) );
253                     
254                     mainHandle.invoke( arguments );
255                 }
256                 catch ( IllegalAccessException e )
257                 { // just pass it on
258                     Thread.currentThread().getThreadGroup().uncaughtException( Thread.currentThread(),
259                                                                                new Exception( "The specified mainClass doesn't contain a main method with appropriate signature.",
260                                                                                               e ) );
261                 }
262                 catch ( InvocationTargetException e )
263                 { // use the cause if available to improve the plugin execution output
264                    Throwable exceptionToReport = e.getCause() != null ? e.getCause() : e;
265                    Thread.currentThread().getThreadGroup().uncaughtException( Thread.currentThread(), exceptionToReport );
266                 }
267                 catch ( Throwable e )
268                 { // just pass it on
269                     Thread.currentThread().getThreadGroup().uncaughtException( Thread.currentThread(), e );
270                 }
271             }
272         }, mainClass + ".main()" );
273         bootstrapThread.setContextClassLoader( getClassLoader() );
274         setSystemProperties();
275 
276         bootstrapThread.start();
277         joinNonDaemonThreads( threadGroup );
278         // It's plausible that spontaneously a non-daemon thread might be created as we try and shut down,
279         // but it's too late since the termination condition (only daemon threads) has been triggered.
280         if ( keepAlive )
281         {
282             getLog().warn( "Warning: keepAlive is now deprecated and obsolete. Do you need it? Please comment on MEXEC-6." );
283             waitFor( 0 );
284         }
285 
286         if ( cleanupDaemonThreads )
287         {
288 
289             terminateThreads( threadGroup );
290 
291             try
292             {
293                 threadGroup.destroy();
294             }
295             catch ( IllegalThreadStateException e )
296             {
297                 getLog().warn( "Couldn't destroy threadgroup " + threadGroup, e );
298             }
299         }
300 
301         if ( originalSystemProperties != null )
302         {
303             System.setProperties( originalSystemProperties );
304         }
305 
306         synchronized ( threadGroup )
307         {
308             if ( threadGroup.uncaughtException != null )
309             {
310                 throw new MojoExecutionException( "An exception occured while executing the Java class. "
311                     + threadGroup.uncaughtException.getMessage(), threadGroup.uncaughtException );
312             }
313         }
314 
315         registerSourceRoots();
316     }

318     /**
319      * a ThreadGroup to isolate execution and collect exceptions.
320      */
321     class IsolatedThreadGroup
322         extends ThreadGroup
323     {
324         private Throwable uncaughtException; // synchronize access to this
325 
326         public IsolatedThreadGroup( String name )
327         {
328             super( name );
329         }
330 
331         public void uncaughtException( Thread thread, Throwable throwable )
332         {
333             if ( throwable instanceof ThreadDeath )
334             {
335                 return; // harmless
336             }
337             synchronized ( this )
338             {
339                 if ( uncaughtException == null ) // only remember the first one
340                 {
341                     uncaughtException = throwable; // will be reported eventually
342                 }
343             }
344             getLog().warn( throwable );
345         }
346     }

348     private void joinNonDaemonThreads( ThreadGroup threadGroup )
349     {
350         boolean foundNonDaemon;
351         do
352         {
353             foundNonDaemon = false;
354             Collection<Thread> threads = getActiveThreads( threadGroup );
355             for ( Thread thread : threads )
356             {
357                 if ( thread.isDaemon() )
358                 {
359                     continue;
360                 }
361                 foundNonDaemon = true; // try again; maybe more threads were created while we were busy
362                 joinThread( thread, 0 );
363             }
364         }
365         while ( foundNonDaemon );
366     }

368     private void joinThread( Thread thread, long timeoutMsecs )
369     {
370         try
371         {
372             getLog().debug( "joining on thread " + thread );
373             thread.join( timeoutMsecs );
374         }
375         catch ( InterruptedException e )
376         {
377             Thread.currentThread().interrupt(); // good practice if don't throw
378             getLog().warn( "interrupted while joining against thread " + thread, e ); // not expected!
379         }
380         if ( thread.isAlive() ) // generally abnormal
381         {
382             getLog().warn( "thread " + thread + " was interrupted but is still alive after waiting at least "
383                 + timeoutMsecs + "msecs" );
384         }
385     }

387     private void terminateThreads( ThreadGroup threadGroup )
388     {
389         long startTime = System.currentTimeMillis();
390         Set<Thread> uncooperativeThreads = new HashSet<Thread>(); // these were not responsive to interruption
391         for ( Collection<Thread> threads = getActiveThreads( threadGroup ); !threads.isEmpty(); threads =
392             getActiveThreads( threadGroup ), threads.removeAll( uncooperativeThreads ) )
393         {
394             // Interrupt all threads we know about as of this instant (harmless if spuriously went dead (! isAlive())
395             // or if something else interrupted it ( isInterrupted() ).
396             for ( Thread thread : threads )
397             {
398                 getLog().debug( "interrupting thread " + thread );
399                 thread.interrupt();
400             }
401             // Now join with a timeout and call stop() (assuming flags are set right)
402             for ( Thread thread : threads )
403             {
404                 if ( !thread.isAlive() )
405                 {
406                     continue; // and, presumably it won't show up in getActiveThreads() next iteration
407                 }
408                 if ( daemonThreadJoinTimeout <= 0 )
409                 {
410                     joinThread( thread, 0 ); // waits until not alive; no timeout
411                     continue;
412                 }
413                 long timeout = daemonThreadJoinTimeout - ( System.currentTimeMillis() - startTime );
414                 if ( timeout > 0 )
415                 {
416                     joinThread( thread, timeout );
417                 }
418                 if ( !thread.isAlive() )
419                 {
420                     continue;
421                 }
422                 uncooperativeThreads.add( thread ); // ensure we don't process again
423                 if ( stopUnresponsiveDaemonThreads )
424                 {
425                     getLog().warn( "thread " + thread + " will be Thread.stop()'ed" );
426                     thread.stop();
427                 }
428                 else
429                 {
430                     getLog().warn( "thread " + thread + " will linger despite being asked to die via interruption" );
431                 }
432             }
433         }
434         if ( !uncooperativeThreads.isEmpty() )
435         {
436             getLog().warn( "NOTE: " + uncooperativeThreads.size() + " thread(s) did not finish despite being asked to "
437                 + " via interruption. This is not a problem with exec:java, it is a problem with the running code."
438                 + " Although not serious, it should be remedied." );
439         }
440         else
441         {
442             int activeCount = threadGroup.activeCount();
443             if ( activeCount != 0 )
444             {
445                 // TODO this may be nothing; continue on anyway; perhaps don't even log in future
446                 Thread[] threadsArray = new Thread[1];
447                 threadGroup.enumerate( threadsArray );
448                 getLog().debug( "strange; " + activeCount + " thread(s) still active in the group " + threadGroup
449                     + " such as " + threadsArray[0] );
450             }
451         }
452     }

454     private Collection<Thread> getActiveThreads( ThreadGroup threadGroup )
455     {
456         Thread[] threads = new Thread[threadGroup.activeCount()];
457         int numThreads = threadGroup.enumerate( threads );
458         Collection<Thread> result = new ArrayList<Thread>( numThreads );
459         for ( int i = 0; i < threads.length && threads[i] != null; i++ )
460         {
461             result.add( threads[i] );
462         }
463         return result; // note: result should be modifiable
464     }

482     /**
483      * Set up a classloader for the execution of the main class.
484      * 
485      * @return the classloader
486      * @throws MojoExecutionException if a problem happens
487      */
488     private ClassLoader getClassLoader()
489         throws MojoExecutionException
490     {
491         List<Path> classpathURLs = new ArrayList<>();
492         this.addRelevantPluginDependenciesToClasspath( classpathURLs );
493         this.addRelevantProjectDependenciesToClasspath( classpathURLs );
494         this.addAdditionalClasspathElements( classpathURLs );
495         
496         try
497         {
498             return LoaderFinder.find( classpathURLs, mainClass );
499         }
500         catch ( NullPointerException | IOException e )
501         {
502             throw new MojoExecutionException( e.getMessage(), e );
503         }
504 
505     }

507     private void addAdditionalClasspathElements( List<Path> path )
508     {
509         if ( additionalClasspathElements != null )
510         {
511             for ( String classPathElement : additionalClasspathElements )
512             {
513                 Path file = Paths.get( classPathElement );
514                 if ( !file.isAbsolute() )
515                 {
516                     file = project.getBasedir().toPath().resolve( file );
517                 }
518                 getLog().debug( "Adding additional classpath element: " + file + " to classpath" );
519                 path.add( file );
520             }
521         }
522     }

561             List<Path> theClasspathFiles = new ArrayList<>();

563             collectProjectArtifactsAndClasspath( artifacts, theClasspathFiles );

565             for ( Path classpathFile : theClasspathFiles )

567                 getLog().debug( "Adding to classpath : " + classpathFile );
568                 path.add( classpathFile );
569             }
570 
571             for ( Artifact classPathElement : artifacts )

573                 getLog().debug( "Adding project dependency artifact: " + classPathElement.getArtifactId()

575                 path.add( classPathElement.getFile().toPath() );

577         }

579         {
580             getLog().debug( "Project Dependencies will be excluded." );
581         }

585     /**
586      * Determine all plugin dependencies relevant to the executable. Takes includePlugins, and the executableDependency
587      * into consideration.
588      * 
589      * @return a set of Artifact objects. (Empty set is returned if there are no relevant plugin dependencies.)
590      * @throws MojoExecutionException if a problem happens resolving the plufin dependencies
591      */
592     private Set<Artifact> determineRelevantPluginDependencies()
593         throws MojoExecutionException
594     {
595         Set<Artifact> relevantDependencies;
596         if ( this.includePluginDependencies )
597         {
598             if ( this.executableDependency == null )
599             {
600                 getLog().debug( "All Plugin Dependencies will be included." );
601                 relevantDependencies = new HashSet<Artifact>( this.pluginDependencies );
602             }
603             else
604             {
605                 getLog().debug( "Selected plugin Dependencies will be included." );
606                 Artifact executableArtifact = this.findExecutableArtifact();
607                 relevantDependencies = this.resolveExecutableDependencies( executableArtifact );
608             }
609         }
610         else
611         {
612             relevantDependencies = Collections.emptySet();
613             getLog().debug( "Plugin Dependencies will be excluded." );
614         }
615         return relevantDependencies;
616     }

618     /**
619      * Resolve the executable dependencies for the specified project
620      * 
621      * @param executablePomArtifact the project's POM
622      * @return a set of Artifacts
623      * @throws MojoExecutionException if a failure happens
624      */
625     private Set<Artifact> resolveExecutableDependencies( Artifact executablePomArtifact )
626         throws MojoExecutionException
627     {
628 
629         Set<Artifact> executableDependencies = new LinkedHashSet<>();
630         try
631         {
632             ProjectBuildingRequest buildingRequest = getSession().getProjectBuildingRequest();
633             
634             MavenProject executableProject =
635                 this.projectBuilder.build( executablePomArtifact, buildingRequest ).getProject();
636 
637             for ( ArtifactResult artifactResult : dependencyResolver.resolveDependencies( buildingRequest, executableProject.getModel(), null ) )
638             {
639                 executableDependencies.add( artifactResult.getArtifact() );
640             }
641         }
642         catch ( Exception ex )
643         {
644             throw new MojoExecutionException( "Encountered problems resolving dependencies of the executable "
645                 + "in preparation for its execution.", ex );
646         }
647 
648         return executableDependencies;
649     }

651     /**
652      * Stop program execution for nn millis.
653      * 
654      * @param millis the number of millis-seconds to wait for, <code>0</code> stops program forever.
655      */
656     private void waitFor( long millis )
657     {
658         Object lock = new Object();
659         synchronized ( lock )
660         {
661             try
662             {
663                 lock.wait( millis );
664             }
665             catch ( InterruptedException e )
666             {
667                 Thread.currentThread().interrupt(); // good practice if don't throw
668                 getLog().warn( "Spuriously interrupted while waiting for " + millis + "ms", e );
669             }
670         }
671     }
