91              return systemProperties.get( key );

95          {
96              return commandLines.size();
97          }

101             return commandLines.get( index );

195     /**
196      * 
197      * @param pom the pom file
198      * @param goal the goal to execute
199      * @return output from System.out during mojo execution
200      * @throws Exception if any exception occurs
201      */
202     protected String execute( File pom, String goal )
203         throws Exception
204     {
205 
206         ExecMojo mojo;
207         mojo = (ExecMojo) lookupMojo( goal, pom );
208 
209         setUpProject( pom, mojo );
210 
211         MavenProject project = (MavenProject) getVariableValueFromObject( mojo, "project" );
212 
213         // why isn't this set up by the harness based on the default-value? TODO get to bottom of this!
214         // setVariableValueToObject( mojo, "includeProjectDependencies", Boolean.TRUE );
215         // setVariableValueToObject( mojo, "killAfter", new Long( -1 ) );
216 
217         assertNotNull( mojo );
218         assertNotNull( project );
219 
220         // trap System.out
221         PrintStream out = System.out;
222         StringOutputStream stringOutputStream = new StringOutputStream();
223         System.setOut( new PrintStream( stringOutputStream ) );
224         // ensure we don't log unnecessary stuff which would interfere with assessing success of tests
225         mojo.setLog( new DefaultLog( new ConsoleLogger( Logger.LEVEL_ERROR, "exec:exec" ) ) );
226 
227         try
228         {
229             mojo.execute();
230         }
231         catch ( Throwable e )
232         {
233             e.printStackTrace( System.err );
234             fail( e.getMessage() );
235         }
236         finally
237         {
238             System.setOut( out );
239         }
240 
241         return stringOutputStream.toString();
242     }

244     private void setUpProject( File pomFile, ExecMojo mojo )
245         throws Exception
246     {

257         ProjectBuilder builder = lookup( ProjectBuilder.class );
258

260 
261         MavenProject project = builder.build( pomFile, buildingRequest ).getProject();
262 
263         // this gets the classes for these tests of this mojo (exec plugin) onto the project classpath for the test
264         project.getBuild().setOutputDirectory( new File( "target/test-classes" ).getAbsolutePath() );
265 
266         mojo.setProject( project );
267 
268         mojo.setLog( new SystemStreamLog()
269         {
270             public boolean isDebugEnabled()
271             {
272                 return true;
273             }
274         } );

277     // MEXEC-12, MEXEC-72
278     public void testGetExecutablePath()
279         throws IOException
280     {
281 
282         ExecMojo realMojo = new ExecMojo();
283 
284         File workdir = new File( System.getProperty( "user.dir" ) );
285         Map<String, String> enviro = new HashMap<String, String>();
286 
287         String myJavaPath = "target" + File.separator + "javax";
288         File f = new File( myJavaPath );
289         assertTrue( "file created...", f.createNewFile() );
290         assertTrue( "file exists...", f.exists() );
291 
292         realMojo.setExecutable( myJavaPath );
293 
294         CommandLine cmd = realMojo.getExecutablePath( enviro, workdir );
295         assertTrue( "File exists so path is absolute",
296                     cmd.getExecutable().startsWith( System.getProperty( "user.dir" ) ) );
297 
298         f.delete();
299         assertFalse( "file deleted...", f.exists() );
300         cmd = realMojo.getExecutablePath( enviro, workdir );
301         assertEquals( "File doesn't exist. Let the system find it (in that PATH?)", myJavaPath, cmd.getExecutable() );
302 
303         if ( OS.isFamilyWindows() ) // Exec Maven Plugin only supports Batch detection and PATH search on Windows
304         {
305             myJavaPath = "target" + File.separator + "javax.bat";
306             f = new File( myJavaPath );
307             assertTrue( "file created...", f.createNewFile() );
308             assertTrue( "file exists...", f.exists() );
309 
310             final String comSpec = System.getenv( "ComSpec" );
311             
312             realMojo.setExecutable( "javax.bat" );
313             cmd = realMojo.getExecutablePath( enviro, workdir );
314             assertTrue( "is bat file on windows, execute using ComSpec.", cmd.getExecutable().equals( comSpec ) );
315 
316             enviro.put( "PATH", workdir.getAbsolutePath() + File.separator + "target" );
317             cmd = realMojo.getExecutablePath( enviro, workdir );
318             assertTrue( "is bat file on windows' PATH, execute using ComSpec.", cmd.getExecutable().equals( comSpec ) );
319             f.delete();
320             assertFalse( "file deleted...", f.exists() );
321         }
322     }

324     // under Windows cmd/bat files should be preferred over files with the same prefix without extension, e.g.
325     // if there are "node" and "node.cmd" the mojo should pick "node.cmd"
326     // see https://github.com/mojohaus/exec-maven-plugin/issues/42
327     public void testGetExecutablePathPreferExecutableExtensionsOnWindows()
328     		throws IOException
329     {
330     	// this test is for Windows
331     	if (!OS.isFamilyWindows()) {
332     		return;
333     	}
334     	final ExecMojo realMojo = new ExecMojo();
335     	
336     	final String tmp = System.getProperty("java.io.tmpdir");
337     	final File workdir = new File(tmp, "testGetExecutablePathPreferExecutableExtensionsOnWindows");
338     	workdir.mkdirs();
339     	
340     	final Map<String, String> enviro = new HashMap<String, String>();
341     	
342     	final File f = new File(workdir, "mycmd");
343     	final File fCmd = new File(workdir, "mycmd.cmd");
344     	f.createNewFile();
345     	fCmd.createNewFile();
346     	assertTrue( "file exists...", f.exists() );
347     	assertTrue( "file exists...", fCmd.exists() );
348     	
349     	realMojo.setExecutable( "mycmd" );
350     	
351     	final CommandLine cmd = realMojo.getExecutablePath( enviro, workdir );
352     	// cmdline argumets are: [/c, %path-to-temp%\mycmd.cmd], so check second argument
353     	assertTrue( "File should have cmd extension",
354     			cmd.getArguments()[1].endsWith( "mycmd.cmd" ) );
355     	
356     	f.delete();
357     	fCmd.delete();
358     	assertFalse( "file deleted...", f.exists() );
359     	assertFalse( "file deleted...", fCmd.exists() );
360     	
361     }

450     private void checkMojo( String expectedCommandLine )
451     {
452         assertEquals( 1, mojo.getAmountExecutedCommandLines() );
453         CommandLine commandline = mojo.getExecutedCommandline( 0 );
454         // do NOT depend on Commandline toString()
455         assertEquals( expectedCommandLine, getCommandLineAsString( commandline ) );
456     }

458     private String getCommandLineAsString( CommandLine commandline )
459     {
460         // for the sake of the test comparisons, cut out the eventual
461         // cmd /c *.bat conversion
462         String result = commandline.getExecutable();
463         boolean isCmd = false;
464         if ( OS.isFamilyWindows() && result.equals( "cmd" ) )
465         {
466             result = "";
467             isCmd = true;
468         }
469         String[] arguments = commandline.getArguments();
470         for ( int i = 0; i < arguments.length; i++ )
471         {
472             String arg = arguments[i];
473             if ( isCmd && i == 0 && "/c".equals( arg ) )
474             {
475                 continue;
476             }
477             if ( isCmd && i == 1 && arg.endsWith( ".bat" ) )
478             {
479                 arg = arg.substring( 0, arg.length() - ".bat".length() );
480             }
481             result += ( result.length() == 0 ? "" : " " ) + arg;
482         }
483         return result;
484     }
