1   package org.codehaus.mojo.exec;
2   
3   import java.util.Timer;
4   import java.util.TimerTask;
5   
6   /**
7    * @author <a href="mailto:dsmiley@mitre.org">David Smiley</a>
8    */
9   public class MainWithThreads
10      extends Thread
11  {
12      public static final String ALL_EXITED = "t1(interrupted td)(cancelled timer)";
13  
14      public static final String TIMER_IGNORED = "t1(interrupted td)";
15  
16      /**
17       * both daemon threads will be interrupted as soon as the non daemon thread is done. the responsive daemon thread
18       * will be interrupted right away. - if the timer is cancelled (using 'cancelTimer' as argument), the timer will die
19       * on itself after all the other threads - if not, one must use a time out to stop joining on that unresponsive
20       * daemon thread
21       * 
22       * @param args the arguments
23       **/
24      public static void main( String... args )
25      {
26          // long run so that we interrupt it before it ends itself
27          Thread responsiveDaemonThread = new MainWithThreads( 60000, "td" );
28          responsiveDaemonThread.setDaemon( true );
29          responsiveDaemonThread.start();
30  
31          new MainWithThreads( 200, "t1" ).start();
32  
33          // Timer in Java <= 6 aren't interruptible
34          final Timer t = new Timer( true );
35  
36          if ( optionsContains( args, "cancelTimer" ) )
37          {
38              t.schedule( new TimerTask()
39              {
40                  public void run()
41                  {
42                      System.out.print( "(cancelled timer)" );
43                      t.cancel();
44                  }
45              }, 400 );
46          }
47      }
48  
49      private static boolean optionsContains( String[] args, String option )
50      {
51          for ( String arg : args )
52          {
53              if ( arg.equals( option ) )
54              {
55                  return true;
56              }
57          }
58          return false;
59      }
60  
61      private int millisecsToSleep;
62  
63      private String message;
64  
65      public MainWithThreads( int millisecsToSleep, String message )
66      {
67          this.millisecsToSleep = millisecsToSleep;
68          this.message = message;
69      }
70  
71      public void run()
72      {
73          try
74          {
75              Thread.sleep( millisecsToSleep );
76          }
77          catch ( InterruptedException e ) // IE's are a way to cancel a thread
78          {
79              System.out.print( "(interrupted " + message + ")" );
80              return;
81          }
82          System.out.print( message );
83      }
84  }
