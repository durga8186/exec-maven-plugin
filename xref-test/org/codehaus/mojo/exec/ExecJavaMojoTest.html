242     /**
243      * @return output from System.out during mojo execution
244      */
245     private String execute( File pom, String goal )
246         throws Exception
247     {
248 
249         ExecJavaMojo mojo;
250         mojo = (ExecJavaMojo) lookupMojo( goal, pom );
251 
252         setUpProject( pom, mojo );
253 
254         MavenProject project = (MavenProject) getVariableValueFromObject( mojo, "project" );
255 
256         // why isn't this set up by the harness based on the default-value? TODO get to bottom of this!
257         setVariableValueToObject( mojo, "includeProjectDependencies", Boolean.TRUE );
258         setVariableValueToObject( mojo, "killAfter", (long) -1 );
259         setVariableValueToObject( mojo, "cleanupDaemonThreads", Boolean.TRUE );
260         setVariableValueToObject( mojo, "classpathScope", "compile" );
261 
262         assertNotNull( mojo );
263         assertNotNull( project );
264 
265         // trap System.out
266         PrintStream out = System.out;
267         StringOutputStream stringOutputStream = new StringOutputStream();
268         System.setOut( new PrintStream( stringOutputStream ) );
269         // ensure we don't log unnecessary stuff which would interfere with assessing success of tests
270         mojo.setLog( new DefaultLog( new ConsoleLogger( Logger.LEVEL_ERROR, "exec:java" ) ) );
271 
272         try
273         {
274             mojo.execute();
275         }
276         finally
277         {
278             // see testUncooperativeThread() for explaination
279             Thread.sleep( 300 ); // time seems about right
280             System.setOut( out );
281         }
282 
283         return stringOutputStream.toString();
284     }

286     private void setUpProject( File pomFile, AbstractMojo mojo )
287         throws Exception
288     {
289         super.setUp();
290         
291         MockitoAnnotations.initMocks( this );
292         
293         ProjectBuildingRequest buildingRequest = mock( ProjectBuildingRequest.class );
294         when( session.getProjectBuildingRequest() ).thenReturn( buildingRequest );
295         MavenRepositorySystemSession repositorySession = new MavenRepositorySystemSession();
296         repositorySession.setLocalRepositoryManager( new SimpleLocalRepositoryManager( LOCAL_REPO ) );
297         when( buildingRequest.getRepositorySession() ).thenReturn( repositorySession );
298         
299         ProjectBuilder builder = lookup( ProjectBuilder.class );
300 
301         MavenProject project = builder.build( pomFile, buildingRequest ).getProject();
302         // this gets the classes for these tests of this mojo (exec plugin) onto the project classpath for the test
303         project.getBuild().setOutputDirectory( new File( "target/test-classes" ).getAbsolutePath() );
304         setVariableValueToObject( mojo, "project", project );
305     }
