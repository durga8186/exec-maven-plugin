195     /**
196      * @return output from System.out during mojo execution
197      */
198     private String execute( File pom, String goal )
199         throws Exception
200     {
201 
202         ExecJavaMojo mojo;
203         mojo = (ExecJavaMojo) lookupMojo( goal, pom );
204 
205         setUpProject( pom, mojo );
206 
207         MavenProject project = (MavenProject) getVariableValueFromObject( mojo, "project" );
208 
209         // why isn't this set up by the harness based on the default-value? TODO get to bottom of this!
210         setVariableValueToObject( mojo, "includeProjectDependencies", Boolean.TRUE );
211         setVariableValueToObject( mojo, "killAfter", (long) -1 );
212         setVariableValueToObject( mojo, "cleanupDaemonThreads", Boolean.TRUE );
213         setVariableValueToObject( mojo, "classpathScope", "compile" );
214 
215         assertNotNull( mojo );
216         assertNotNull( project );
217 
218         // trap System.out
219         PrintStream out = System.out;
220         StringOutputStream stringOutputStream = new StringOutputStream();
221         System.setOut( new PrintStream( stringOutputStream ) );
222         // ensure we don't log unnecessary stuff which would interfere with assessing success of tests
223         mojo.setLog( new DefaultLog( new ConsoleLogger( Logger.LEVEL_ERROR, "exec:java" ) ) );
224 
225         try
226         {
227             mojo.execute();
228         }
229         finally
230         {
231             // see testUncooperativeThread() for explaination
232             Thread.sleep( 300 ); // time seems about right
233             System.setOut( out );
234         }
235 
236         return stringOutputStream.toString();
237     }
238 
239     private void setUpProject( File pomFile, AbstractMojo mojo )
240         throws Exception
241     {
242         MavenProjectBuilder builder = (MavenProjectBuilder) lookup( MavenProjectBuilder.ROLE );
243 
244         ArtifactRepositoryLayout localRepositoryLayout =
245             (ArtifactRepositoryLayout) lookup( ArtifactRepositoryLayout.ROLE, "default" );
246 
247         String path = "src/test/repository";
248 
249         ArtifactRepository localRepository =
250             new DefaultArtifactRepository( "local", "file://" + new File( path ).getAbsolutePath(),
251                                            localRepositoryLayout );
252 
253         MavenProject project = builder.buildWithDependencies( pomFile, localRepository, null );
254         // this gets the classes for these tests of this mojo (exec plugin) onto the project classpath for the test
255         project.getBuild().setOutputDirectory( new File( "target/test-classes" ).getAbsolutePath() );
256         setVariableValueToObject( mojo, "project", project );
257     }
