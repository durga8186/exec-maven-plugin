51      static class MockExecMojo
52          extends ExecMojo
53      {
54          public int executeResult;
55  
56          public List<CommandLine> commandLines = new ArrayList<CommandLine>();
57  
58          public String failureMsg;
59  
60          public Map systemProperties = new HashMap();
61  
62          protected int executeCommandLine( Executor exec, CommandLine commandLine, Map enviro, OutputStream out,
63                                            OutputStream err )
64                                                throws IOException, ExecuteException
65          {
66              commandLines.add( commandLine );
67              if ( failureMsg != null )
68              {
69                  throw new ExecuteException( failureMsg, executeResult );
70              }
71              return executeResult;
72          }
73  
74          protected String getSystemProperty( String key )
75          {
76              return (String) systemProperties.get( key );
77          }
78  
79          int getAmountExecutedCommandLines()
80          {
81              return commandLines.size();
82          }
83  
84          CommandLine getExecutedCommandline( int index )
85          {
86              return ( (CommandLine) commandLines.get( index ) );
87          }
88      }

182     /**
183      * @return output from System.out during mojo execution
184      */
185     private String execute( File pom, String goal )
186         throws Exception
187     {
188 
189         ExecMojo mojo;
190         mojo = (ExecMojo) lookupMojo( goal, pom );
191 
192         setUpProject( pom, mojo );
193 
194         MavenProject project = (MavenProject) getVariableValueFromObject( mojo, "project" );
195 
196         // why isn't this set up by the harness based on the default-value? TODO get to bottom of this!
197         // setVariableValueToObject( mojo, "includeProjectDependencies", Boolean.TRUE );
198         // setVariableValueToObject( mojo, "killAfter", new Long( -1 ) );
199 
200         assertNotNull( mojo );
201         assertNotNull( project );
202 
203         // trap System.out
204         PrintStream out = System.out;
205         StringOutputStream stringOutputStream = new StringOutputStream();
206         System.setOut( new PrintStream( stringOutputStream ) );
207         // ensure we don't log unnecessary stuff which would interfere with assessing success of tests
208         mojo.setLog( new DefaultLog( new ConsoleLogger( Logger.LEVEL_ERROR, "exec:exec" ) ) );
209 
210         try
211         {
212             mojo.execute();
213         }
214         catch ( Throwable e )
215         {
216             e.printStackTrace( System.err );
217             fail( e.getMessage() );
218         }
219         finally
220         {
221             System.setOut( out );
222         }
223 
224         return stringOutputStream.toString();
225     }

227     private void setUpProject( File pomFile, ExecMojo mojo )
228         throws Exception
229     {
230         MavenProjectBuilder builder = (MavenProjectBuilder) lookup( MavenProjectBuilder.ROLE );
231 
232         ArtifactRepositoryLayout localRepositoryLayout =
233             (ArtifactRepositoryLayout) lookup( ArtifactRepositoryLayout.ROLE, "default" );
234 
235         String path = "src/test/repository";
236 
237         ArtifactRepository localRepository =
238             new DefaultArtifactRepository( "local", "file://" + new File( path ).getAbsolutePath(),
239                                            localRepositoryLayout );
240 
241         mojo.setBasedir( File.createTempFile( "mvn-temp", "txt" ).getParentFile() );
242 
243         MavenProject project = builder.buildWithDependencies( pomFile, localRepository, null );
244 
245         // this gets the classes for these tests of this mojo (exec plugin) onto the project classpath for the test
246         project.getBuild().setOutputDirectory( new File( "target/test-classes" ).getAbsolutePath() );
247 
248         mojo.setProject( project );
249 
250         mojo.setLog( new SystemStreamLog()
251         {
252             public boolean isDebugEnabled()
253             {
254                 return true;
255             }
256         } );
257     }

259     // MEXEC-12, MEXEC-72
260     public void testGetExecutablePath()
261         throws IOException
262     {
263 
264         ExecMojo realMojo = new ExecMojo();
265 
266         File workdir = new File( System.getProperty( "user.dir" ) );
267         Map enviro = new HashMap();
268 
269         String myJavaPath = "target" + File.separator + "javax";
270         File f = new File( myJavaPath );
271         assertTrue( "file created...", f.createNewFile() );
272         assertTrue( "file exists...", f.exists() );
273 
274         realMojo.setExecutable( myJavaPath );
275 
276         CommandLine cmd = realMojo.getExecutablePath( enviro, workdir );
277         assertTrue( "File exists so path is absolute",
278                     cmd.getExecutable().startsWith( System.getProperty( "user.dir" ) ) );
279 
280         f.delete();
281         assertFalse( "file deleted...", f.exists() );
282         cmd = realMojo.getExecutablePath( enviro, workdir );
283         assertEquals( "File doesn't exist. Let the system find it (in that PATH?)", myJavaPath, cmd.getExecutable() );
284 
285         if ( OS.isFamilyWindows() ) // how to make this part of the test run on other platforms as well??
286         {
287 
288             myJavaPath = "target" + File.separator + "javax.bat";
289             f = new File( myJavaPath );
290             assertTrue( "file created...", f.createNewFile() );
291             assertTrue( "file exists...", f.exists() );
292 
293             realMojo.setExecutable( "javax.bat" );
294             cmd = realMojo.getExecutablePath( enviro, workdir );
295             assertTrue( "is bat file on windows, execute using cmd.", cmd.getExecutable().equals( "cmd" ) );
296 
297             enviro.put( "PATH", workdir.getAbsolutePath() + File.separator + "target" );
298             cmd = realMojo.getExecutablePath( enviro, workdir );
299             assertTrue( "is bat file on windows' PATH, execute using cmd.", cmd.getExecutable().equals( "cmd" ) );
300             f.delete();
301             assertFalse( "file deleted...", f.exists() );
302         }
303     }

363     public void testIsResultCodeAFailure()
364     {
365         ExecMojo execMojo = new ExecMojo();
366         assertTrue( execMojo.isResultCodeAFailure( 1 ) );
367         assertFalse( execMojo.isResultCodeAFailure( 0 ) );
368 
369         execMojo.setSuccessCodes( new Integer[0] );
370         assertTrue( execMojo.isResultCodeAFailure( 1 ) );
371         assertFalse( execMojo.isResultCodeAFailure( 0 ) );
372 
373         execMojo.setSuccessCodes( new Integer[] { Integer.valueOf( "2" ), Integer.valueOf( "5" ) } );
374         assertTrue( execMojo.isResultCodeAFailure( 0 ) );
375         assertTrue( execMojo.isResultCodeAFailure( 10 ) );
376         assertFalse( execMojo.isResultCodeAFailure( 2 ) );
377         assertFalse( execMojo.isResultCodeAFailure( 5 ) );

392     private void checkMojo( String expectedCommandLine )
393     {
394         assertEquals( 1, mojo.getAmountExecutedCommandLines() );
395         CommandLine commandline = mojo.getExecutedCommandline( 0 );
396         // do NOT depend on Commandline toString()
397         assertEquals( expectedCommandLine, getCommandLineAsString( commandline ) );
398     }
399 
400     private String getCommandLineAsString( CommandLine commandline )
401     {
402         // for the sake of the test comparisons, cut out the eventual
403         // cmd /c *.bat conversion
404         String result = commandline.getExecutable();
405         boolean isCmd = false;
406         if ( OS.isFamilyWindows() && result.equals( "cmd" ) )
407         {
408             result = "";
409             isCmd = true;
410         }
411         String[] arguments = commandline.getArguments();
412         for ( int i = 0; i < arguments.length; i++ )
413         {
414             String arg = arguments[i];
415             if ( isCmd && i == 0 && "/c".equals( arg ) )
416             {
417                 continue;
418             }
419             if ( isCmd && i == 1 && arg.endsWith( ".bat" ) )
420             {
421                 arg = arg.substring( 0, arg.length() - ".bat".length() );
422             }
423             result += ( result.length() == 0 ? "" : " " ) + arg;
424         }
425         return result;
426     }
